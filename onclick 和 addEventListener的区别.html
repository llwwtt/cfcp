onclick事件会覆盖 而addEventListener都会执行


var btn =document.getElementById("btn");
btn.onclick=function(){
alert("点击按钮触发的事件onclick")
}
btn.onclick=function(){
alert("第二次定义的onclik事件")
}
btn.addEventListener("click",function(){
alert("第一次触发addEventListener事件");
});
btn.addEventListener("click",function(){
alert("第二次触发addEventListener事件");
});
